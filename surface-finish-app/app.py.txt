import streamlit as st
import numpy as np
import pandas as pd

st.title("Surface Finish Quality Monitoring System")

def compute_fft(signal, fs):
    n = len(signal)
    freq = np.fft.rfftfreq(n, d=1/fs)
    fft_vals = np.abs(np.fft.rfft(signal))
    return freq, fft_vals

def extract_features(signal, fs):
    freq, fft_vals = compute_fft(signal, fs)

    peak_amp = np.max(fft_vals)
    dominant_freq = freq[np.argmax(fft_vals)]
    rms = np.sqrt(np.mean(signal**2))

    return {
        "Peak Amplitude": float(peak_amp),
        "Dominant Frequency": float(dominant_freq),
        "RMS": float(rms)
    }

def classify_surface(features):
    peak = features["Peak Amplitude"]
    freq = features["Dominant Frequency"]

    if peak < 70 and freq < 55:
        return "Excellent (Grinding Quality)"
    elif peak < 90:
        return "Good"
    elif peak < 110:
        return "Medium"
    else:
        return "Poor (Turning-like Surface)"

st.write("Upload vibration CSV file")

file = st.file_uploader("Upload CSV", type=["csv"])
fs = st.number_input("Sampling Frequency", value=5000)

if file is not None:
    data = pd.read_csv(file)
    signal = data.iloc[:, 0].values

    features = extract_features(signal, fs)
    result = classify_surface(features)

    st.json(features)
    st.success(result)
